\section{The Harsh Truths \& Core Techniques}
% =============================================================================

\noindent\textbf{The Harsh Truths of Competitive Programming}
\begin{sidebar}
\begin{itemize}[leftmargin=1.5em, itemsep=5pt]

  \item \textbf{Typing is not problem-solving:} If your fingers are moving, your
        brain is usually turned off. Spend 80\% of your time on paper, 20\% on
        the keyboard.

  \item \textbf{Easy problems are a trap:} Solving 1{,}000 easy problems gives
        you a fake dopamine hit. If you aren't struggling and your brain doesn't
        physically hurt trying to bridge a logical gap, you aren't growing.

  \item \textbf{CSES is the wrench, Codeforces is the engine:} CSES taught you
        algorithms. Codeforces will hand you scrap metal and ask you to build a
        machine. Do not force heavy CSES data structures onto simple CF
        ad-hoc math problems.

  \item \textbf{Embrace the Flatline:} Your rating will flatline for months.
        You will feel stupid. Then, overnight, a pattern will click, and you
        will jump 200 points. Do not quit.

\end{itemize}
\end{sidebar}

\vspace{4pt}
\noindent\textbf{The Golden Techniques}
\begin{sidebar}
\begin{itemize}[leftmargin=1.5em, itemsep=5pt]

  \item \textbf{Reverse-Engineer Constraints:} Always look at $N$ before the
        story.
        \begin{itemize}
          \item If $N \le 2{,}000$, then $O(N^2)$ is safe.
          \item If $N \le 2 \cdot 10^5$, you are strictly bound to
                $O(N \log N)$ or $O(N)$.
        \end{itemize}

  \item \textbf{The ``Three Strikes'' Rule:} Never open an editorial unless you
        have written down at least \emph{three distinct, failed approaches} on
        paper.

  \item \textbf{The ``Amnesia'' Rule:} If you read an editorial, close it, step
        away for 30 minutes, and then code it from memory. If you can't, you
        didn't learn it.

\end{itemize}
\end{sidebar}

\section{The Micro-Strategy: The 60-Minute Rule}
% =============================================================================

Apply this strict timeline to \textbf{every single problem} you solve outside of live contests.

\begin{sidebar}
\begin{itemize}[leftmargin=1.5em, itemsep=6pt]

  \item \textbf{Minutes 0--5 (Reconnaissance):} Read the problem twice. Check
        constraints. Guess the intended time complexity.

  \item \textbf{Minutes 5--25 (Observation):} \textbf{No keyboard.} Pen and
        paper only. Draw sample cases. Find the math/logic invariant.

  \item \textbf{Minutes 25--35 (Attack It):} Try to break your logic with
        extreme edge cases (e.g., $N = 1$, all zeros, sorted array).

  \item \textbf{Minutes 35--45 (Code It):} Only touch the keyboard if your
        logic survived your own attacks.

  \item \textbf{Minutes 45--60 (The Struggle):} Debug or rethink.

  \item \textbf{Minute 60+ (Editorial Protocol):} Read \emph{only the first
        hint}. Close the tab. Wait 30 minutes. Code from memory.

\end{itemize}
\end{sidebar}

\subsection{Morning: The Logic Awakening}
% ─────────────────────────────────────────────────────────────────────────────

\begin{itemize}[leftmargin=2em, itemsep=6pt]

  \item \textbf{05:00 AM -- 05:30 AM $|$ System Boot (30 min):} Wake up, drink
        1\,L of water. Zero screens. Walk outside or stretch to wake up your
        brain.

  \item \textbf{05:30 AM -- 09:30 AM $|$ Deep Logic \& Observation (4 hrs):}
    \begin{itemize}
      \item \textbf{Target:} USACO Section-Wise (Silver/Gold).
      \item \textbf{Execution:} USACO requires deep thought. Expand the
            micro-strategy to 90 minutes per problem. Spend up to 45 minutes
            purely on paper.
    \end{itemize}

  \item \textbf{09:30 AM -- 10:00 AM $|$ Refuel (30 min):} Quick breakfast.
        Step away from the desk.

\end{itemize}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Mid-Day: The Classic Foundations}
% ─────────────────────────────────────────────────────────────────────────────

\begin{itemize}[leftmargin=2em, itemsep=6pt]

  \item \textbf{10:00 AM -- 01:00 PM $|$ CF Foundations --- Part A (3 hrs):}
    \begin{itemize}
      \item \textbf{Target:} C2Ladders or CF ``Most Solved''
            (+100 to +200 rating above current level).
      \item \textbf{Execution:} Apply the strict 60-Minute Micro-Strategy.
            Learn the classic CP tricks.
    \end{itemize}

  \item \textbf{01:00 PM -- 02:00 PM $|$ Hard Reset (1 hr):} Lunch and a
        mandatory 30--40 minute nap. Your brain is running on low sleep; this
        nap is non-negotiable for afternoon survival.

  \item \textbf{02:00 PM -- 03:30 PM $|$ CF Foundations --- Part B (1.5 hrs):}
    \begin{itemize}
      \item \textbf{Target:} Finish your C2Ladders quota for the day.
    \end{itemize}

\end{itemize}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Afternoon: Targeted Repair}
% ─────────────────────────────────────────────────────────────────────────────

\begin{itemize}[leftmargin=2em, itemsep=6pt]

  \item \textbf{03:30 PM -- 07:30 PM $|$ Weakness Elimination (4 hrs):}
    \begin{itemize}
      \item \textbf{Target:} Topic-wise Upsolve
            (e.g., Graph DP, Binary Search, Constructives).
      \item \textbf{Execution:} Apply the 60-Minute Micro-Strategy. Since you
            know the topic tag, spend minutes 5--25 figuring out how the
            algorithm is hidden in the story.
    \end{itemize}

  \item \textbf{07:30 PM -- 08:00 PM $|$ Pre-Contest Disconnect (30 min):}
        Fast dinner. Completely detach from CP to clear your mind before the
        contest window.

\end{itemize}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Evening: The Contest Arena}
% ─────────────────────────────────────────────────────────────────────────────

\begin{itemize}[leftmargin=2em, itemsep=6pt]

  \item \textbf{08:00 PM -- 10:30 PM $|$ The Modern Meta \& Live Contests
        (2.5 hrs):}
    \begin{itemize}
      \item \textbf{Target:} Live Codeforces Rounds
            (usually 8:05 PM or 8:35 PM BD Time).
      \item \textbf{Execution:} If there is a live contest, compete with zero
            distractions. If there is no live contest, run a Virtual Contest
            from recent Div.\,2 / Div.\,3 rounds. Ignore the 60-minute rule
            for problems A and B to practice raw speed.
    \end{itemize}

\end{itemize}

% ─────────────────────────────────────────────────────────────────────────────
\subsection{Night: Codification \& Shutdown}
% ─────────────────────────────────────────────────────────────────────────────

\begin{itemize}[leftmargin=2em, itemsep=6pt]

  \item \textbf{10:30 PM -- 11:30 PM $|$ Codification (1 hr):}
    \begin{itemize}
      \item \textbf{Target:} Post-Contest Upsolve \& Team Reference Document
            (TRD) Build.
      \item \textbf{Execution:} Take the code from the contest you just did.
            Clean it up, read the editorial for the problem you failed, and add
            any new templates to your TRD.
    \end{itemize}

  \item \textbf{11:30 PM $|$ Hard Shutdown:} Screens off. Go to sleep
        immediately.

\end{itemize}

\section{Most Important Words for Competitive Programming}
% =============================================================================

\begin{sidebarGray}
\begin{itemize}[leftmargin=1.5em, itemsep=6pt]

  \item \textbf{Upsolve:} Solving problems after the contest ends. Contests
        measure your skill; upsolving \emph{builds} it.

  \item \textbf{Paper:} Your true compiler. A wrong answer on paper costs
        0 penalty points.

  \item \textbf{Constraints:} The blueprint. The constraints whisper the
        intended algorithm to you before you even read the story.

  \item \textbf{Counter-Example:} The act of playing devil's advocate and
        aggressively trying to break your own logic before typing.

  \item \textbf{Occam's Razor:} The simplest solution is usually the right one.
        If a 1400-rated problem requires 150 lines of complex data structures,
        your approach is wrong.

\end{itemize}
\end{sidebarGray}

\section{Problem Solving Tricks \& Growth Strategy}
% =============================================================================

\subsection*{When Should I Move to the Next Rating Bracket?}

\begin{itemize}[leftmargin=2em, itemsep=4pt]

  \item \textbf{The ``Rule of 10'':} Do not blindly solve 300 problems per
        rating bracket. If you can solve \emph{10 problems in a row} at a
        specific rating (e.g., 1400) within 45 minutes each, \emph{without
        ever looking at the editorial}, you have mastered that rating.
        Move up immediately.

\end{itemize}

\subsection*{Strategy for CF Rating Growth}

\begin{itemize}[leftmargin=2em, itemsep=6pt]

  \item \textbf{Stop doing easy problems in practice:} Jump straight to
        problems C and D during your CF Foundation blocks. Get comfortable
        feeling completely lost.

  \item \textbf{Hunt your weaknesses:} Look at your last 5 contest
        performances. Identify the exact tag that stopped you, and spend your
        03:30 PM block destroying that specific weakness for a whole week.

  \item \textbf{Read the Standings:} During a live contest, if you are stuck
        on Problem C but 1{,}000 people have solved Problem D, abandon C
        immediately. The scoreboard is telling you that D is secretly easier.

\end{itemize}
